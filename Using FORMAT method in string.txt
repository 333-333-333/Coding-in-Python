#The format() method
# Using Curly braces
print("{} and {} both are besties".format("Mahak","Manas"))
#output: Mahak and Manas both are the best friend
#Positional Argument
print("{1} and {0} best players".format("Mahak","Manas"))
#Keyword Argument
print("{a},{b},{c}".format(a = "Mahak", b = "Manas", c = "Ritik"))
